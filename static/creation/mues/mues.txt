        var rectHeight = 20;
        var rectLeft = (window.innerWidth - rectWidth) / 3.8;
        var rectTop = (window.innerHeight - rectHeight) / 7;

        // Écouter l'événement "mousemove" sur la page
        document.addEventListener("mousemove", function (event) {
            // Vérifier si les coordonnées du curseur sont dans le rectangle centré sur la page
            if (event.pageX > rectLeft && event.pageX < rectLeft + rectWidth &&
                event.pageY > rectTop && event.pageY < rectTop + rectHeight) {
                // Si le curseur est dans le rectangle, changer le curseur en image personnalisée
                document.body.style.cursor = "url(''), auto";
            } else {
                // Si le curseur est en dehors du rectangle, utiliser le curseur par défaut
